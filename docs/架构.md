HPL项目根目录/
├── src/                    # 源代码目录
│   ├── models.py          # 数据模型定义（类、对象、函数、AST节点）
│   ├── lexer.py           # 词法分析器（Token生成）
│   ├── parser.py          # 顶层解析器（YAML解析、文件包含处理）
│   ├── ast_parser.py      # AST解析器（语法分析，生成抽象语法树）
│   ├── evaluator.py       # 执行器（AST执行、表达式求值）
│   └── interpreter.py     # 主入口点（命令行接口）
├── examples/              # 示例文件目录
│   ├── base.hpl           # 基础类定义示例
│   ├── example.hpl        # 主示例程序
│   ├── test_fixes.hpl     # 功能测试示例
│   ├── test_for_loop.hpl  # 循环测试示例
│   ├── test_comment_comprehensive.hpl  # 注释测试示例
│   └── test_low_priority.hpl         # 高级特性测试示例
├── docs/                  # 文档目录
│   ├── 架构.md            # 项目架构说明（本文件）
│   └── HPL语法手册.md      # 语法手册
├── tests/                 # 测试目录
│   └── test_*.py          # 单元测试文件（已添加）
├── .gitignore             # Git忽略文件
├── LICENSE                # 许可证
├── README.md              # 项目说明
├── requirements.txt       # Python依赖
└── run_example.py         # 示例运行脚本
